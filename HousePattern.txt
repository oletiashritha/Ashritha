public class HousePattern{
public static void main(String args[]){
int n=5;
for(int i=1;i<=n;i++){
for(int s=n;s>i;s--){
System.out.println(" ");
}
for(int j=1;j<=(2*i-1);j++){
if(j==1||j==(2*i-1)||i==n)
System.outprintln("*");
else
System.out.println(" ");
}
System.out.println();
}
for(int i=1;i<=n-1;i++){
for(int j=1;j<=(2*n-1);j++){
if(j==1||j==(2*n-1)||i==n-1)
System.out.println("*");
else
System.out.println(" ");
}
System.out.println();
}
}
}
